254 public void UpdateGroupAssignments()
255 {
256 var keys = GetComponentsInChildren<Collider>(true)
257 .Where(x => !x.isTrigger && ReferenceEquals(this, x.GetComponentInParent<IHealth>()))
258 .Select(x => x.gameObject)
259 .Distinct();
260
261 // Remove gameObjects that are no longer children
262 var objectsToRemove = m_groupAssignments.Keys.Except(keys).ToArray();
263 foreach (var key in objectsToRemove)
264 {
265 m_groupAssignments.Remove(key);
266 }
267
268 // Add gameObjects that are new children
269 foreach (var key in keys.Except(m_groupAssignments.Keys))
270 {
271 m_groupAssignments.Add(key, null);
272 }
