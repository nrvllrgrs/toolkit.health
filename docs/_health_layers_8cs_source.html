182 private void Health_Died(HealthEventArgs e)
183 {
184 if (m_indexMap.TryGetValue(e.hit.victim as IHealth, out int index))
185 {
186 var current = m_layers[index];
187
188 // Advance index to next living health
189 int i = index + 1;
190 for (; i < m_layers.Length; ++i)
191 {
192 if (!m_layers[i].health.isDead)
193 {
194 m_index = i;
195 if (current.overflowDamage)
196 {
197 float overflowDamage = e.hit.value - e.delta;
198 m_layers[i].health.Apply(new DamageHit(overflowDamage, e.hit.damageType, e.hit));
199 }
200 break;
201 }
202 }
203
204 SetIsDead(i >= m_layers.Length);
205 }
206
207 // Notify after index has been updated
208 m_onDied?.Invoke(e);
209 }
